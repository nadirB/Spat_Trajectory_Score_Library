@global $pi
$pi := 3.141592653589793

@fun_def X($t, $offsetX) 

{ 
	@cos($t) + $offsetX
} 

@fun_def Y( $t, $offsetY) 
{ 
	
	;$r*(($q-1)*@sin($t)) - @sin(($q-1)*$t) + $offsetY

}

{ 
	
	;$r*(($q-1)*@sin($t)) - @sin(($q-1)*$t) + $offsetY

}
