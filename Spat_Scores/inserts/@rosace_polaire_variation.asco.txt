@global $pi 
$pi := 3.141592653589793

@fun_def X($t, $p, $q) 
{ 
	($p*@cos($t)+@cos($q*$t))
} 

@fun_def Y($t,$p,$q) 
{ 
	($p*@sin($t)-@sin($q*$t))	 
}
