
;La rosace d'équation polaire r(θ) = a cos(kθ + const.
; x=cos(kt)sin(t)
;y = cos(kt)cos(t) 

;k réel
; k rationnel courbe fermée et finie
;k irrationnel courbe non fermée et infinie

@fun_def X($t, $k) 

{ 

	@cos($k*$t)*@sin($t)
} 

/////////////////

@fun_def Y($t, $k) 
{ 
	

	@sin($k*$t)*@cos($t)
}

@proc_def ::rosace_trajectory($cmd, $initT, $targeT, $k, $SPEED)

{	@global $pi
	@local $SPEED

	$pi := 3.141592653589793
	
	

	;loop pi $SPEED
	;{
		abort picurve

		curve picurve @grain := 0.01s 
						@Action := {
								
								@command($cmd) xy @X($t, $k) @Y($t, $k)
							}

		

		
		
		{
			$t
					{
							   { ($initT*$pi) } /*@type "exp"*/
						$SPEED { ($targeT*$pi) }
						
	
					}
		}


	}

;}
