@proc_def ::cardinalX_trajectory($cmd, $initX, $X, $func, $cycle, $amp, $translate, $SPEED)

{	
	@global $SPEED ; for setvar...

	@local $pi
	$pi := 3.141592653589793

	;$x ≠ 0 !
	
	

	;loop pi $SPEED
	;{
		abort sinicurve

		curve sinicurve @grain := 0.01s 
						@Action := {
								$y := ($func(($cycle*$pi)*$x)/($pi*$x))*($amp)+($translate) ; tweaked sinc function
								@command($cmd) $x $y 
							}

		{
			$x
					{
								{ $initX  } @type "sine_in_out" 
						 $SPEED { $X }
						
	
					}
		}


	;}

}
