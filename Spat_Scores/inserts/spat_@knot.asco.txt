@global $pi
	$pi := 3.141592653589793

@global $m, $offsetX, $offsetX
$m := 1





@fun_def X( $t, $m, $offsetX) 
{ 
	

	@tan($m*$t) * @cos($t) + $offsetX
}



@fun_def Y($t, $m, $offsetY) 

{ 

	@tan($m*$t) * @sin($t) + $offsetY
} 



/////////////////

{

	@local $offset

    ;loop pi $SPEED
	;{

						@Action := {
								
								@command($cmd) xy @X($t, $m, $offsetX) @Y($t, $m, $offsetY)
							}

		

		
		
		{
			$t
					{
							   { ($initT*$pi)  } /*@type "exp"*/
						$SPEED { ($endT*$pi) }
						
	
					}
		}


	;}


}
