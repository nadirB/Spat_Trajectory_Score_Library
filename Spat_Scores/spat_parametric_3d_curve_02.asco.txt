x = sin(3t)cos(5t)
y = sin(3t)sin(5T)
z = cos(3t)



*/






@global $pi
	$pi := 3.141592653589793






@fun_def X($t, $offsetX) 

{ 

	

	@sin(3*$t)*@cos(5*$t) + $offsetX
} 

/////////////////

@fun_def Y( $t, $offsetY) 
{ 
	


	@sin(3*$t)*@sin(5*$t) + $offsetY

}


@fun_def Z( $t, $offsetZ) 
{ 
	


	@cos(3*$t)  + $offsetZ

}

							}

		

		
		
		{
			$t
					{
							   { ($initT*$pi)  } /*@type "exp"*/
						$speed { ($endT*$pi) }
						
	
					}
		}


	;}

}
